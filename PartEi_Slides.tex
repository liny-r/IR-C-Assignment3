% ---- 3. KM vs AJ contrast ---------------------------------------------------
\begin{frame}{Two estimators, two different treatments of competing events.}
    \begin{columns}[T,onlytextwidth]
        \column{0.48\textwidth}
        \begin{block}{Kaplan--Meier}
        \[
            \hat{F}^{(k)}_{\mathrm{KM}}(t)
            = 1 - \prod_{t_j \le t}\!\left(1 - \frac{d_{j}^{(k)}}{n_j^{\mathrm{KM}}}\right)
        \]
        \vspace{0.1em}
        \small
        \begin{tabular}{@{}ll@{}}
            $d_j^{(k)}$ & cause-$k$ exits at $t_j$ \\[2pt]
            $n_j^{\mathrm{KM}}$ & risk set \textcolor{bad}{excluding} competing-event loans \\[2pt]
        \end{tabular}
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{Aalen--Johansen}
        \[
            \hat{F}_k(t) = \sum_{t_j \le t} \hat{S}(t_j^{-})\cdot \frac{d_{kj}}{n_j}
        \]
        \vspace{0.1em}
        \small
        \begin{tabular}{@{}ll@{}}
            $d_{kj}$        & cause-$k$ exits at $t_j$ \\[2pt]
            $n_j$           & risk set \textcolor{good}{including} all exits \\[2pt]
            $\hat{S}(t_j^-)$ & overall survival just before $t_j$ \\[2pt]
        \end{tabular}
        \end{block}
    \end{columns}
    \vspace{0.6em}
    \begin{center}
        \small
        \textcolor{bad}{KM}: competing exits censored $\Rightarrow$ denominator too small $\Rightarrow$ each increment inflated.\\[3pt]
        \textcolor{good}{AJ}: $\hat{S}(t_j^-)$ shrinks with every exit $\Rightarrow$ $\hat{F}^{(\mathrm{pre})} + \hat{F}^{(\mathrm{def})} + \hat{S} = 1$ always.
    \end{center}
\end{frame}

% ---- 8. Key Takeaways ----------------------------------------------------
\begin{frame}{Key Takeaways}
    \begin{enumerate}\itemsep5pt
        \item \textbf{Competing risks are not optional}: 4 covariates flip sign between causes.
        \item \textbf{KM bias reaches $+$2.2\,pp at 20 years}: coherent forecasting requires Aalen--Johansen.
        \item \textbf{Vintage dominates}: 2007 default CIF is $6\times$ long-run average; post-2020 prepayment is frozen.
        \item \textbf{FICO separates default $6\times$; it barely separates prepayment}.
        \item \textbf{Rate incentive is the dominant prepayment signal with zero default content}: $e^{0.256} \approx 1.29$.
        \item \textbf{Fine--Gray $\approx$ cause-specific Cox here}; the distinction is material in subprime or crisis pools.
    \end{enumerate}
\end{frame}
