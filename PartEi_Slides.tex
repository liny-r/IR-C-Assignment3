% ---- 9. Key Takeaways ----------------------------------------------------
\begin{frame}{Key Takeaways}
    \begin{enumerate}\itemsep5pt
        \item \textbf{Competing risks are not optional}: 4 covariates flip sign between causes.
        \item \textbf{KM bias reaches $+$2.2\,pp at 20 years}: coherent forecasting requires Aalen--Johansen.
        \item \textbf{Vintage dominates}: 2007 default CIF is $6\times$ long-run average; post-2020 prepayment is frozen.
        \item \textbf{FICO separates default $6\times$; it barely separates prepayment}.
        \item \textbf{Rate incentive is the dominant prepayment signal with zero default content}: $e^{0.256} \approx 1.29$.
        \item \textbf{Fine--Gray $\approx$ cause-specific Cox here}; the distinction is material in subprime or crisis pools.
    \end{enumerate}
\end{frame}
